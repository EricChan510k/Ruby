def search
	@doctors = Doctor.search params[:search]
	respond_to do |format|
	  format.json {
	      render :json => {
		      :success => true,
		      :records_found => @doctors.count,
		      :data => {:doctors => @doctors.map{|doctor| {:first_name => doctor.first_name,:last_name => doctor.last_name} } } 
		    }
      }   
    end  	
end
